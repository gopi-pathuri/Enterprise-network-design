! dist-router-1.txt
version 15.4
service timestamps debug datetime msec
service timestamps log datetime msec
hostname Dist-R1
!
interface GigabitEthernet0/0
 description Uplink to Core-SW1
 ip address 10.0.0.2 255.255.255.252
 ip ospf 1 area 0
 duplex auto
 speed auto
!
interface GigabitEthernet0/1.10
 description VLAN 10 Management
 encapsulation dot1Q 10
 ip address 10.1.10.2 255.255.255.0
 standby 1 ip 10.1.10.1
 standby 1 priority 110
 standby 1 preempt
!
interface GigabitEthernet0/1.20
 description VLAN 20 HR
 encapsulation dot1Q 20
 ip address 10.1.20.2 255.255.255.0
 standby 2 ip 10.1.20.1
 standby 2 priority 110
 standby 2 preempt
 ip access-group HR_ACL in
!
interface GigabitEthernet0/1.30
 description VLAN 30 Sales
 encapsulation dot1Q 30
 ip address 10.1.30.2 255.255.255.0
 standby 3 ip 10.1.30.1
 standby 3 priority 110
 standby 3 preempt
!
router ospf 1
 router-id 2.2.2.2
 redistribute connected subnets
!
ip access-list extended HR_ACL
 permit tcp 10.1.20.0 0.0.0.255 10.1.10.0 0.0.0.255 eq 22
 deny ip any any
!
end
